## 1. Schema Migration

### 1.1 New columns on `predictions`

```sql
-- Migration: add to schema.sql and run as a guarded migration in db.ts

ALTER TABLE predictions ADD COLUMN model_probability REAL;
ALTER TABLE predictions ADD COLUMN market_probability REAL;
ALTER TABLE predictions ADD COLUMN outcome_basis TEXT DEFAULT 'legacy'
  CHECK (outcome_basis IN ('final', 'estimated', 'legacy'));

-- Indexes for metrics queries
CREATE INDEX IF NOT EXISTS idx_predictions_outcome_basis
  ON predictions (outcome_basis);

CREATE INDEX IF NOT EXISTS idx_predictions_learning
  ON predictions (outcome_basis, domain, outcome_timestamp DESC)
  WHERE outcome_basis = 'final'
    AND model_probability IS NOT NULL
    AND market_probability IS NOT NULL
    AND outcome IS NOT NULL;
```

**Backfill rule:** All existing rows keep `outcome_basis = 'legacy'` (the DEFAULT). No backfill of `model_probability` or `market_probability` — those values are unknowable for legacy rows.

### 1.2 `learning_examples` view

Add to `schema.sql` after the `predictions` table definition:

```sql
CREATE VIEW IF NOT EXISTS learning_examples AS
SELECT
  id,
  domain,
  regime_tag           AS regime,
  strategy_class,
  symbol,
  model_probability,
  market_probability,
  executed,
  position_size,
  CASE WHEN outcome = 'YES' THEN 1 ELSE 0 END                              AS outcome_value,
  pnl,
  (model_probability  - CASE WHEN outcome = 'YES' THEN 1.0 ELSE 0.0 END)
  * (model_probability  - CASE WHEN outcome = 'YES' THEN 1.0 ELSE 0.0 END) AS brier_model,
  (market_probability - CASE WHEN outcome = 'YES' THEN 1.0 ELSE 0.0 END)
  * (market_probability - CASE WHEN outcome = 'YES' THEN 1.0 ELSE 0.0 END) AS brier_market,
  created_at,
  outcome_timestamp    AS resolved_at
FROM predictions
WHERE outcome_basis     = 'final'
  AND model_probability  IS NOT NULL
  AND market_probability IS NOT NULL
  AND outcome            IS NOT NULL;
```

### 2.4 Where does `decision.probability` come from?

In v1.97+, the LLM originator returns a structured proposal that includes a probability estimate. That value must be threaded through:

```
LlmOriginator.propose() → TradeCandidate.probability
  → LlmEntryGate.evaluate() preserves it (no mutation)
    → execution call site reads candidate.probability
      → createPrediction({ modelProbability: candidate.probability })
```

Check `src/core/llm_entry_gate.ts` and the originator output type to confirm `probability` is present on the candidate object passed to the execution layer. If it is missing, add it — this is part of this task.

## 4. Metrics Engine — `src/memory/learning_metrics.ts` (new file)

```typescript
import { openDatabase } from './db.js';

export type WindowMetrics = {
  windowSize: number;
  sampleCount: number;
  accuracy: number | null;
  brierModel: number | null;
  brierMarket: number | null;
  brierDelta: number | null;   // brierMarket - brierModel; positive = Thufir beats market
  avgModelProbability: number | null;
  avgMarketProbability: number | null;
  avgEdge: number | null;
  totalPnl: number | null;
};

const WINDOWS = [10, 20, 50, 100, 200] as const;
const MIN_SAMPLE = 20;

export function computeRollingWindowMetrics(domain?: string): WindowMetrics[] {
  const db = openDatabase();

  // Fetch last 200 final examples once; slice in JS for each window
  const domainFilter = domain ? `AND domain = ?` : '';
  const params = domain ? [200, domain] : [200];

  const rows = db.prepare(`
    SELECT
      outcome_value,
      brier_model,
      brier_market,
      model_probability,
      market_probability,
      pnl
    FROM learning_examples
    ${domain ? 'WHERE domain = ?' : ''}
    ORDER BY resolved_at DESC
    LIMIT 200
  `).all(...(domain ? [domain] : [])) as Array<{
    outcome_value: number;
    brier_model: number;
    brier_market: number;
    model_probability: number;
    market_probability: number;
    pnl: number | null;
  }>;

  return WINDOWS.map((size) => {
    const slice = rows.slice(0, size);
    const n = slice.length;

    if (n < MIN_SAMPLE) {
      return { windowSize: size, sampleCount: n, accuracy: null, brierModel: null,
               brierMarket: null, brierDelta: null, avgModelProbability: null,
               avgMarketProbability: null, avgEdge: null, totalPnl: null };
    }

    let correct = 0, brierModelSum = 0, brierMarketSum = 0;
    let modelProbSum = 0, marketProbSum = 0, pnlSum = 0;

    for (const row of slice) {
      const predicted = row.model_probability >= 0.5 ? 1 : 0;
      if (predicted === row.outcome_value) correct++;
      brierModelSum  += row.brier_model;
      brierMarketSum += row.brier_market;
      modelProbSum   += row.model_probability;
      marketProbSum  += row.market_probability;
      pnlSum         += row.pnl ?? 0;
    }

    const brierModel  = brierModelSum  / n;
    const brierMarket = brierMarketSum / n;

    return {
      windowSize:           size,
      sampleCount:          n,
      accuracy:             correct / n,
      brierModel,
      brierMarket,
      brierDelta:           brierMarket - brierModel,
      avgModelProbability:  modelProbSum  / n,
      avgMarketProbability: marketProbSum / n,
      avgEdge:              (modelProbSum - marketProbSum) / n,
      totalPnl:             pnlSum,
    };
  });
}

export function computeDomainWindowMetrics(): Record<string, WindowMetrics[]> {
  const db = openDatabase();
  const domains = db.prepare(
    `SELECT DISTINCT domain FROM learning_examples WHERE domain IS NOT NULL`
  ).all() as Array<{ domain: string }>;

  const result: Record<string, WindowMetrics[]> = {};
  for (const { domain } of domains) {
    result[domain] = computeRollingWindowMetrics(domain);
  }
  return result;
}
```

### 5.2 Pre-check in `evaluate()`

Before the LLM call, run a deterministic calibration check:

```typescript
function calibrationBlock(domain: string | undefined): {
  blocked: boolean;
  sizeMultiplier: number;
  reason: string | null;
} {
  if (!domain) return { blocked: false, sizeMultiplier: 1, reason: null };

  const windows = computeRollingWindowMetrics(domain);
  const w50 = windows.find((w) => w.windowSize === 50);
  const w20 = windows.find((w) => w.windowSize === 20);

  // Block domain if Brier delta is negative over 50 samples
  if (w50 && w50.sampleCount >= 50 && w50.brierDelta !== null && w50.brierDelta < 0) {
    return { blocked: true, sizeMultiplier: 0, reason: 'domain_calibration_below_market' };
  }

  // Reduce size if Brier delta is degrading over 20 samples
  if (w20 && w20.sampleCount >= 20 && w20.brierDelta !== null && w20.brierDelta < -0.05) {
    return { blocked: false, sizeMultiplier: 0.5, reason: 'domain_calibration_degrading' };
  }

  return { blocked: false, sizeMultiplier: 1, reason: null };
}
```

Apply in `evaluate()` before the LLM call:

```typescript
const cal = calibrationBlock(candidate.domain);
if (cal.blocked) {
  return { verdict: 'reject', reason: cal.reason, sizeMultiplier: 0 };
}
// ... existing gate logic ...
// After gate approves, apply sizeMultiplier to suggested risk %
```

function buildPredictionAccuracySection(): {
  global: WindowMetrics[];
  byDomain: Record<string, WindowMetrics[]>;
  totalFinalPredictions: number;
} {
  try {
    const db = openDatabase();
    const total = (db.prepare(
      `SELECT COUNT(*) AS c FROM learning_examples`
    ).get() as { c: number }).c;

    return {
      global:               computeRollingWindowMetrics(),
      byDomain:             computeDomainWindowMetrics(),
      totalFinalPredictions: total,
    };
  } catch {
    return { global: [], byDomain: {}, totalFinalPredictions: 0 };
  }
}

## 8. Test Requirements

### Phase 1 tests

- `createPrediction` with `modelProbability` and `marketProbability` — fields persisted correctly
- `createPrediction` without `modelProbability` — row created, excluded from `learning_examples`
- `recordOutcome` with `outcomeBasis: 'final'` — row appears in `learning_examples`
- `recordOutcome` with `outcomeBasis: 'estimated'` — row excluded from `learning_examples`
- Legacy row (no `outcome_basis` set) — excluded from `learning_examples`

### Phase 2 tests

- `computeRollingWindowMetrics()` with < 20 rows — all metric fields null
- `computeRollingWindowMetrics()` with 25 rows — window-10 and window-20 populated, window-50+ null
- `brierDelta` is positive when model probability is closer to outcome than market probability
- `calibrationBlock` returns `blocked: true` when domain has 50+ samples and negative Brier delta
- `calibrationBlock` returns `sizeMultiplier: 0.5` when domain has 20+ samples and Brier delta < -0.05
- `calibrationBlock` returns `blocked: false` when domain has < 20 samples (insufficient data path)
